a. 2 3/10

[tex]5 \frac{1}{2} - \frac{4}{5} \times 4[/tex]

[tex] = \frac{5 \times 2 + 1}{2} - \frac{4}{5} \times 4[/tex]

[tex] = \frac{11}{2} - \frac{4}{5} \times \frac{4}{1} [/tex]

[tex] = \frac{11}{2} - \frac{4 \times 4}{5 \times 1} [/tex]

[tex] = \frac{11}{2} - \frac{16}{5} [/tex]

[tex] = \frac{23}{10} = 2 \frac{3}{10} [/tex]
